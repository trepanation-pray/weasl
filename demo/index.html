weasl.js

Javascript based pop up modal.

Usage

Add

Add to you project:

yarn

yarn add @trepanation-pray/weasl

npm

npm install @trepanation-pray/weasl

Include

Include the files in your project.

css

@import 'node_module/@trepanation-pray/weasl/dist/weasl.min.css';

scss

@import 'node_module/@trepanation-pray/weasl/src/weasl.scss';

javascript

@import 'node_module/@trepanation-pray/weasl/dist/weasl.min.js';

Use

Using the function as is will fire the modal on page load.

weasl({
  // Unique name will be applied to modal as a class, no spaces (Required)
  name: 'name-of-modal',
  // Sets the content of the modal, create or get this and pass in. (Required)
  content: yourContent,
  // Sets an additional class(es) on the modal (Optional)
  style: 'success',
  // Overwrite default padding (Optional)
  padding: '50px',
  // Overwrite default max-width (Optional)
  width: '600px'
});

Example Use

jQuery click

$('.button').click(function() {
  weasl({
    name: 'alert',
    content: 'This is an alert!'
  });
});

Use with cookies.

add cookie helpers

yarn add @trepanation-pray/cookie

include cookie helpers in your build

'node_modules/@trepanation-pray/cookie/index.js'

// Check if cookie is set
if( !getCookie( 'hello' ) ) {
  // If not set run modal
  var yourContent = 'Hello World'
  weasl({
    name: 'hello',
    content: yourContent,
  });
}
// Listen for modal close event
document.addEventListener('weaslClosed', function (e) {
  // Set cookie to acknowledge the popup has been seen.
  setCookie('hello', 'acknowledged pop up', 1, '/');
}, false);

Events

// Dispatched when modal is initialised
document.addEventListener('weaslInit', function (e) {
  console.log('Initialised');
}, false);
// Dispatched when modal has opened
document.addEventListener('weaslOpened', function (e) {
  console.log('Opened');
}, false);
// Dispatched when modal has closed
document.addEventListener('weaslClosed', function (e) {
  console.log('Closed');
}, false);
